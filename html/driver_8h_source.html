98static inline int packet_ring_init(packet_ring_t *r, uint32_t slot_count, int use_blocking)
99{
100 if (!r)
101 return EINVAL;
102 if (!is_power_of_two_u32(slot_count))
103 return EINVAL;
104
105 memset(r, 0, sizeof(*r));
106 r->slots = (packet_slot_t *)calloc(slot_count, sizeof(packet_slot_t));
107 if (!r->slots)
108 return ENOMEM;
109
110 r->slot_count = slot_count;
111 r->mask = slot_count - 1;
112 r->use_blocking = use_blocking ? 1 : 0;
113
114 pthread_mutex_init(&r->mu, NULL);
115 pthread_cond_init(&r->not_empty, NULL);
116 pthread_cond_init(&r->not_full, NULL);
117 return 0;
118}

120static inline void packet_ring_destroy(packet_ring_t *r)
121{
122 if (!r)
123 return;
124 pthread_mutex_destroy(&r->mu);
125 pthread_cond_destroy(&r->not_empty);
126 pthread_cond_destroy(&r->not_full);
127 free(r->slots);
128 memset(r, 0, sizeof(*r));
129}

131/*------------------------ enqueue / dequeue (copy-in/cout-out) --------------*/
132static inline int packet_ring_enq(packet_ring_t *r,
133 const uint8_t *data,
134 uint32_t len,
135 uint64_t ts_ns)
136{
137 if (!r || (!data && len))
138 return EINVAL;
139 if (len > PACKET_MAX_BYTES)
140 return EMSGSIZE;
141
142 pthread_mutex_lock(&r->mu);
143
144 while (r->count == r->slot_count)
145 {
146 r->drop_full++;
147 if (!r->use_blocking)
148 {
149 pthread_mutex_unlock(&r->mu);
150 return EAGAIN; // full
151 }
152 pthread_cond_wait(&r->not_full, &r->mu);
153 }
154 packet_slot_t *s = &r->slots[r->tail & r->mask];
155 s->len = len;
156 s->ts_ns = ts_ns;
157 if (len)
158 memcpy(s->data, data, len);
159
160 r->tail++;
161 r->count++;
162 r->enq_ok++;
163
164 pthread_cond_signal(&r->not_empty);
165 pthread_mutex_unlock(&r->mu);
166 return 0;
167}

169static inline int packet_ring_deq(packet_ring_t *r,
170 uint8_t *out,
171 uint32_t out_cap,
172 uint32_t *out_len,
173 uint64_t *out_ts_ns)
174{
175 if (!r || !out_len)
176 return EINVAL;
177 pthread_mutex_lock(&r->mu);
178
179 while (r->count == 0)
180 {
181 if (!r->use_blocking)
182 {
183 pthread_mutex_unlock(&r->mu);
184 return EAGAIN;
185 }
186 pthread_cond_wait(&r->not_empty, &r->mu);
187 }
188
189 packet_slot_t *s = &r->slots[r->head & r->mask];
190 uint32_t len = s->len;
191
192 if (out)
193 {
194 if (out_cap < len)
195 {
196 pthread_mutex_unlock(&r->mu);
197 return EMSGSIZE;
198 }
199 if (len)
200 memcpy(out, s->data, len);
201 }
202 *out_len = len;
203 if (out_ts_ns)
204 *out_ts_ns = s->ts_ns;
205
206 r->head++;
207 r->count--;
208 r->deq_ok++;
209
210 pthread_cond_signal(&r->not_full);
211 pthread_mutex_unlock(&r->mu);
212 return 0;
213}

217static inline int packet_queue_set_init(packet_queue_set_t *set,
218 uint32_t packet_queue_count,
219 uint32_t slot_count, // 기본 4096,
220 int user_blocking)
221{
222 if (!set)
223 return EINVAL;
224 if (packet_queue_count < MIN_QUEUE_COUNT || packet_queue_count > MAX_QUEUE_COUNT)
225 return EINVAL;
226 if (slot_count == 0)
227 slot_count = DEFAULT_SLOT_COUNT;
228 if (!is_power_of_two_u32(slot_count))
229 return EINVAL;
230
231 memset(set, 0, sizeof(*set));
232 set->q = (packet_ring_t *)calloc(packet_queue_count, sizeof(packet_ring_t));
233 if (!set->q)
234 return ENOMEM;
235
236 set->qcount = packet_queue_count;
237 for (uint32_t i = 0; i < packet_queue_count; i++)
238 {
239 int rc = packet_ring_init(&set->q[i], slot_count, user_blocking);
240 if (rc != 0)
241 {
242 for (uint32_t j = 0; j < i; j++)
243 packet_ring_destroy(&set->q[j]);
244 free(set->q);
245 memset(set, 0, sizeof(*set));
246 return rc;
247 }
248 }
249
250 return 0;
251}
252
253static inline void packet_queue_set_destroy(packet_queue_set_t *set)
254{
255 if (!set || !set->q)
256 return;
257 for (uint32_t i = 0; i < set->qcount; i++)
258 packet_ring_destroy(&set->q[i]);
259 free(set->q);
260 memset(set, 0, sizeof(*set));
261}
